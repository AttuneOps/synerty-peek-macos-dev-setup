macVersion=$(sw_vers -productVersion)

Step 4 - Install Xcode and utilities

Step 4.1 - Check if Xcode is installed

Execute the following script: 

                
if [[ ! -d /Applications/Xcode.app ]]; then
    echo "Please install Xcode manually"
    exit 1
else
    echo "Xcode is installed manually."
fi

Step 4.2 - Setup Xcode

Execute the following script: 

                
# Reset the development directory path to put to Xcode
#  /Applications/Xcode.app/Contents/Developer

sudo xcode-select -r

This script may require you to answer the following prompts:

Prompt Answer
password: {MacSSHUser.password}

Execute the following script: 

                
#!/bin/bash

Step 6 - Install and Setup Homebrew

Step 6.1 - Install Homerbew to local user

Execute the following script: 

                
if [[ -d ${HOME}/opt/homebrew  ]]
then
    echo "homebrew already installed"
    exit 0
fi

mkdir -p ~/opt
cd ~/opt

mkdir homebrew

curl -L https://github.com/Homebrew/brew/tarball/master \
    | tar xz --strip 1 -C homebrew

Step 6.2 - Setup Homebrew

Execute the following script: 

                
cat <<EOF >> ~/.bash_profile
export PATH="`echo ~/opt/homebrew/bin`:\$PATH"
EOF

source ~/.bash_profile

Step 6.3 - Install gnu-sed

Execute the following script: 

                
brew install gnu-sed

Step 6.4 - Setup gnu-sed

Execute the following script: 

                
rm ${HOME}/bin/sed || true
mkdir -p ${HOME}/bin
ln -s `which gsed` ${HOME}/bin/sed

Step 6.5 - Install wget

Execute the following script: 

                
brew install wget

Step 6.6 - Install the dev libs to compile Python

Execute the following script: 

                
brew install openssl@1.1 zlib openldap freetds

Step 6.7 - Install libgeos

Execute the following script: 

                
wget https://raw.githubusercontent.com/Homebrew/homebrew-core/master/Formula/geos.rb -O geos.rb
brew install geos.rb
rm geos.rb

Step 7 - Prepare .bash_profile

Execute the following script: 

                
# ${xyz} is escaped as \${xyz}

cat <<EOF >> ~/.bash_profile
##### SET THE PEEK ENVIRONMENT #####
# Setup the variables for PYTHON
export PEEK_PY_VER="3.9.1"
export PATH="/Users/peek/opt/bin:\$PATH"
export PATH="/Users/peek/bin:\$PATH"

# Set the variables for the platform release
# These are updated by the deploy script
export PEEK_ENV=""
export PATH="\${PEEK_ENV}/bin:\$PATH"
EOF

source ~/.bash_profile

Step 8 - Install Python 3.9.1 from source

Step 8.1 - Deploy Python source tarball

Execute the following script: 

                
cd ~
source .bash_profile
wget https://github.com/python/cpython/archive/v${PEEK_PY_VER}.zip
unzip v${PEEK_PY_VER}.zip

Step 8.2 - Configure the build

Execute the following script: 

                
source ~/.bash_profile
cd cpython-${PEEK_PY_VER}

export LDFLAGS="-L$(brew --prefix)/opt/openssl@1.1/lib -L$(brew --prefix)/opt/zlib/lib"
export CPPFLAGS="-I$(brew --prefix)/opt/openssl@1.1/include -I$(brew --prefix)/opt/zlib/include"
export PKG_CONFIG_PATH="$(brew --prefix)/opt/openssl@1.1/lib/pkgconfig:$(brew --prefix)/opt/zlib/lib/pkgconfig"

./configure \
    --prefix=$HOME/opt \
    --enable-optimizations \
    --enable-shared \
    --with-openssl=$(brew --prefix openssl@1.1)

Step 8.3 - Compile the source

Execute the following script: 

                
cd cpython-${PEEK_PY_VER}
make -j`sysctl -n hw.ncpu`

Step 8.4 - Install the build

Execute the following script: 

                
cd cpython-${PEEK_PY_VER}
make install

Step 8.5 - Cleanup the download and build dir

Execute the following script: 

                
cd
rm -rf cpython-${PEEK_PY_VER}
rm v${PEEK_PY_VER}.zip

Step 8.6 - Make python and pip as a global command

Execute the following script: 

                
cd /Users/peek/opt/bin
rm -f pip python
ln -s pip3 pip
ln -s python3 python
cd

Step 8.7 - Test the setup

Execute the following script: 

                
source ~/.bash_profile

pass="/Users/peek/opt/bin/python"
[ "`which python`" == "$pass" ] && echo "python passed" || (echo "python FAILED - $(which python)" && exit 1)

pass="Python ${PEEK_PY_VER}"
[ "`python --version`" == "$pass" ] && echo "python version passed" || (echo "python version FAILED" && exit 1)

pass="/Users/peek/opt/bin/pip"
[ "`which pip`" == "$pass" ] && echo "pip passed" || (echo "pip FAILED - $(which pip)" && exit 1)


pass="pip 22.0.4 from /Users/peek/opt/lib/python3.9/site-packages/pip (python 3.9)"
[ "`pip --version`" == "$pass" ] && echo "pip version passed" || (echo "pip version FAILED - $(pip --version)" && exit 1)

Step 9 - Install PostgresSQL 12.5 with Python support from source

Step 9.1 - Deploy PostgreSQL source tarball

Execute the following script: 

                
PEEK_PG_VER=12.5
SRC_DIR="$HOME/postgresql-${PEEK_PG_VER}"

# Remove the src dir and install file
rm -rf ${SRC_DIR} || true
cd $HOME

wget https://ftp.postgresql.org/pub/source/v${PEEK_PG_VER}/postgresql-${PEEK_PG_VER}.tar.bz2
tar xjf postgresql-${PEEK_PG_VER}.tar.bz2

cd ${SRC_DIR}

Step 9.2 - Compile and install the build

Execute the following script: 

                
# attune aborts ssh connection with the ./configure and make

PEEK_PG_VER=12.5
SRC_DIR="$HOME/postgresql-${PEEK_PG_VER}"
cd ${SRC_DIR}


export CPPFLAGS=" -I$(brew --prefix openssl@1.1)/include "
export LDFLAGS=" -L$(brew --prefix openssl@1.1)/lib "

echo "configuring source"
# redirect to null to avaid data longer than 10kb to stdout
./configure \
      --disable-debug \
      --prefix=$HOME/opt \
      --enable-thread-safety \
      --with-openssl \
      --with-python > /dev/null

echo "compiling source"
make -j`sysctl -n hw.ncpu` > /dev/null

make install-world

# this is required for timescale to compile
cp ${SRC_DIR}/src/test/isolation/pg_isolation_regress ~/opt/bin

Step 9.3 - Clean up source folder and build folder

Execute the following script: 

                
PEEK_PG_VER=12.5
SRC_DIR="$HOME/postgresql-${PEEK_PG_VER}"

# Remove the src dir and install file
cd
rm -rf ${SRC_DIR}*

Step 9.4 - Initialise PostgreSQL database

Execute the following script: 

                
#Refresh .bash_profile so initdb can find postgres
source .bash_profile

initdb --pgdata=$HOME/pgdata/12 --auth-local=trust  --auth-host=md5

Step 9.5 - Tune the postgresql.conf

Execute the following script: 

                
sed -i 's/max_connections = 100/max_connections = 200/g' ~/pgdata/12/postgresql.conf

Step 11 - Install PostgreSQL Timescaledb

Step 11.1 - Deploy Timescale source

Execute the following script: 

                
PEEK_TSDB_VER=1.7.4

cd
rm -r timescaledb-${PEEK_TSDB_VER} || true
wget https://github.com/timescale/timescaledb/archive/${PEEK_TSDB_VER}.zip
unzip ${PEEK_TSDB_VER}.zip
cd timescaledb-${PEEK_TSDB_VER}

Step 11.2 - Compile and install Timescale

Execute the following script: 

                
PEEK_TSDB_VER=1.7.4
cd timescaledb-${PEEK_TSDB_VER}

export CPPFLAGS=`pg_config --cppflags`
export LDFLAGS=`pg_config --ldflags`
export OPENSSL_ROOT_DIR="$(brew --prefix)/opt/openssl@1.1/"


# Bootstrap the build system
./bootstrap -DAPACHE_ONLY=1

# To build the extension
cd build && make

# To install
make install

# Cleanup the source code
cd
rm -rf ${PEEK_TSDB_VER}.zip
rm -rf timescaledb-${PEEK_TSDB_VER}

Step 11.3 - Install timescaledb-tune

Execute the following script: 

                
cd

rm -rf timescaledb-tools || true
rm timescaledb-tools-0.8.1.tar.lzma || true

wget https://timescalereleases.blob.core.windows.net/homebrew/timescaledb-tools-0.8.1.tar.lzma
tar xf timescaledb-tools-0.8.1.tar.lzma
cp timescaledb-tools/timescaledb-tune $HOME/opt/bin
rm -rf timescaledb-tools
rm timescaledb-tools-0.8.1.tar.lzma

Step 11.4 - Tune with Timescaledb

Execute the following script: 

                
PGVER=12
FILE="$HOME/pgdata/${PGVER}/postgresql.conf"
timescaledb-tune -quiet -yes -conf-path ${FILE} -pg-version ${PGVER}

Step 11.5 - Make PostgreSQL a service

Execute the following script: 

                
# Make sure there is a LaunchAgents folder
mkdir -p ~/Library/LaunchAgents/

# Create the plist file
cat <<EOF > ~/Library/LaunchAgents/postgresql.plist
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>KeepAlive</key>
  <true/>
  <key>Label</key>
  <string>postgresql</string>
  <key>ProgramArguments</key>
  <array>
    <string>${HOME}/opt/bin/postgres</string>
    <string>-D</string>
    <string>${HOME}/pgdata/12</string>
  </array>
  <key>RunAtLoad</key>
  <true/>
  <key>WorkingDirectory</key>
  <string>${HOME}/opt</string>
  <key>StandardOutPath</key>
  <string>${HOME}/pgdata/12/postgres.log</string>
  <key>StandardErrorPath</key>
  <string>${HOME}/pgdata/12/postgres.log</string>
</dict>
</plist>
EOF

# Tell launchctl to use the plist
launchctl load ~/Library/LaunchAgents/postgresql.plist

Step 11.6 - Check that the service has started

Execute the following script: 

                
pgrep -laf postgres

Step 11.7 - Allow OS user peek for passwordless login

Connect via ssh:

ssh {macsshroot}@{macsshhost}

Execute the following script: 

                
sudo mkdir -p /usr/local/var/postgres/
F=/usr/local/var/postgres/pg_hba.conf
cat | sudo tee $F <<EOF
# TYPE  DATABASE        USER            ADDRESS                 METHOD
local   all             postgres                                peer
local   all             peek                                    trust

# "local" is for Unix domain socket connections only
local   all             all                                     peer
# IPv4 local connections:
host    all             all             127.0.0.1/32            md5
# IPv6 local connections:
host    all             all             ::1/128                 md5
EOF

Step 11.8 - Create database peek

Connect via ssh:

ssh {macsshuser}@{macsshhost}

Execute the following script: 

                
createuser peek -S
createdb -O peek peek

Step 11.9 - Set the PostgreSQL peek users password

Execute the following script: 

                
psql peek <<EOF
ALTER ROLE peek WITH PASSWORD 'PASSWORD';
EOF


# Set the password as "PASSWORD" for development machines
# Set it to a secure password from https://xkpasswd.net/s/ for production

# Cleanup traces of the password
[ ! -e ~/.psql_history ] || rm ~/.psql_history

Step 14 - Change max file opens
